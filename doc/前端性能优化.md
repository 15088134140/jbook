## 减少DOM操作
```js
var ul = document.querySelector('ul');

// bad
for(var i = 0; i < 100; i++) {
  var li = document.createElement('li');
  ul.appendChild(li);
}

// good
var fragment = document.createDocumentFragment();
for(var i = 0; i < 100; i++) {
  var li = document.createElement('li');
  fragment.appendChild(li);
}
ul.appendChild(fragment);

// good
var lis = []
for(var i = 0; i < 100; i++) {
  var li = document.createElement('li');
  lis.push(li);
}
$(ul).append(lis);
```
